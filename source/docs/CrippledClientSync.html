Cripple Client

ICS 360 moves data as XML, not JSON. Also, there are legacy issues at play that require sending a lot of custom headers to display error messaging, etc.

We're monkey patching Backbone.sync to apply those headers

 
  CrippledClientSync = (method, model, options) ->

    methodMap =
      'create' : 'POST'
      'update' : 'PUT'
      'delete' : 'DELETE'
      'read'   : 'GET'

    type = methodMap[method]

!! SUPER IMPORTANT !! If you don't spec the Accept for application/xml then you will not get the correct version of Policy XML files

 
      xhr.setRequestHeader('Accept', 'application/xml') 

      if options.basic_auth_digest
        xhr.setRequestHeader('X-Authorization', "Basic #{options.basic_auth_digest}")
        xhr.setRequestHeader('Authorization', "Basic #{options.basic_auth_digest}")
