Calculate the page jump option tags

 
    calculate_pagejumps : ->
      pages = Math.round(+@pagination.total_items / +@pagination.per_page)
      selects = ""
      for page in [1..pages]
        selected = ''
        if page is @pagination.page
          selected = ' selected'
        selects += """
            <option value="#{page}"#{selected}>#{page}</option>
          """
          
      selects
